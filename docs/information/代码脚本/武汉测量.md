```
% 读取bin文件，提取CIR和GPS信息
% 输入:
%   bin_file_path - bin文件路径
%   target_lat - 目标点纬度（可选，用于时延校正）
%   target_lon - 目标点经度（可选，用于时延校正）
% 输出:
%   CIR - 原始CIR矩阵 (frm_num x L_CIR)
%   GPS_data - 结构体包含 lat, lon, alt
%   shifted_CIR - 时延校正后的CIR矩阵

%% 参数设置
U = 1024;
p = 15;  % 每帧数据长度p*1024
L_CIR = 300;  % CIR长度

%% 读取bin文件
fip = fopen(bin_file_path, 'rb');
if fip == -1
    error('无法打开文件: %s', bin_file_path);
end
result_imu = fread(fip);
fclose(fip);

%% 解析帧结构
frem_len = result_imu(6)*256 + result_imu(7);
all_len = length(result_imu);
frm_num = fix(all_len/frem_len);
result_imu_1 = result_imu(1:frem_len*frm_num, 1);
result_iq = reshape(result_imu_1, frem_len, frm_num);

fprintf('读取到 %d 帧数据\n', frm_num);

%% 提取IQ数据
data_i = zeros(frm_num, 1024*p);
data_q = zeros(frm_num, 1024*p);

for frm_cnt = 1:frm_num
    for i = 1:1024*p
        data_q(frm_cnt,i) = result_iq(32+4*(i-1)+1, frm_cnt)*256 + result_iq(32+4*(i-1)+2, frm_cnt);
        data_i(frm_cnt,i) = result_iq(32+4*(i-1)+3, frm_cnt)*256 + result_iq(32+4*(i-1)+4, frm_cnt);

        % 归一化
        if data_i(frm_cnt,i) > 32767
            data_i(frm_cnt,i) = (data_i(frm_cnt,i) - 2^16)/32767;
        else
            data_i(frm_cnt,i) = data_i(frm_cnt,i)/32767;
        end

        if data_q(frm_cnt,i) > 32767
            data_q(frm_cnt,i) = (data_q(frm_cnt,i) - 2^16)/32767;
        else
            data_q(frm_cnt,i) = data_q(frm_cnt,i)/32767;
        end
    end
end

```
